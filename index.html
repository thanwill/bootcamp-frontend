function exponencial(array, expoente = 2) {
    const resultado = [];
    
    for(let i = 0; i < array.length; i++) {
        resultado.push(array[i] ** expoente);
    }

    return resultado;
}

Objeto "Arguments"

Um array com todos os parâmetros passados quando a função foi invocada.


function findMax(){
    // define max como o menor valor possível
    let max = -Infinity;
    // percorre todos os argumentos
    for(let i = 0; i < arguments.length; i++){
        if(arguments[i] > max){
            max = arguments[i];
        }
    }
    return max;
    // findMax(1, 123, 500, 115, 44, 88);
    // 500
}

Arrays

Spread: uma forma de lidar separadamente com elementos. O que era para ser um array se torna um elemento independente.


function sum(x, y, z){
    return x + y + z;
}
const numbers = [1, 2, 3];
console. log(sum(...numbers)); // 6

os três pontinhos antes do array fazem com que cada elemento do array seja passado como um argumento separado para a função
