function exponencial(array, expoente = 2) {
    const resultado = [];
    
    for(let i = 0; i < array.length; i++) {
        resultado.push(array[i] ** expoente);

Objeto "Arguments"

Um array com todos os parâmetros passados quando a função foi invocada.


function findMax(){
    // define max como o menor valor possível
    let max = -Infinity;
    // percorre todos os argumentos
    for(let i = 0; i < arguments.length; i++){
        if(arguments[i] > max){
            max = arguments[i];
        }
    }
    return max;
    // findMax(1, 123, 500, 115, 44, 88);
    // 500
}

Arrays

Spread: uma forma de lidar separadamente com elementos. O que era para ser um array se torna um elemento independente.

Os três pontinhos antes do array fazem com que cada elemento do array seja passado como um argumento separado para a função


function sum(x, y, z){
    return x + y + z;
}
const numbers = [1, 2, 3];
console. log(sum(...numbers)); // 6

Rest: combina vários elementos (argumentos) em um array.

O que era um elemento independente se torna parte de um array.


function confereTamanho(...args){
    console. log(args.length);
}

confereTamanho(); // 0
confereTamanho(1, 2, 3); // 3
confereTamanho(1, "a", true, {a: 1, b: 2}); // 4

Loops

For: é um loop dentro de elementos interáveis (arrays, strings)


function multiplicaPorDois(array){
    let multiplicados = [];
    for(let i = 0; i < array.length; i++){
        multiplicados.push(array[i] * 2);
    }
    return multiplicados;
}
const meuArray = [1, 2, 3];
console.log(multiplicaPorDois(meuArray)); // [2, 4, 6]

For ... in: é um loop entre propriedades enumeráveis de um objeto. 


funtion loopObjeto(objeto){
    for(let prop in objeto){
        console.log(prop);
        // console.log(objeto[prop]);
    }
}
const meuObjeto = {
    nome: "Dio",
    idade: 5,
    cidade: "Rio de Janeiro"
}
loopObjeto(meuObjeto);
// nome, idade, cidade

For ... of : loop entre estruturas interáveis (arrays, strings)


function loopArray(array){
    for(let valor of array){
        console.log(valor);
    }
}
const meuArray = [1, 2, 3];
loopArray(meuArray);
// 1, 2, 3

While: loop que executa enquanto a condição for verdadeira. 


function loopWhile(array){
    let i = 0;
    while(i < array.length){
        console.log(array[i]);
        i++;
    }
}
const meuArray = [1, 2, 3];
loopWhile(meuArray);
// 1, 2, 3

Do ... while: loop que executa primeiro e depois verifica a condição. Mas sempre executará pelo menos uma vez. 


function loopDoWhile(array){
    let i = 0;
    do{
        console.log(array[i]);
        i++;
    } while(i < array.length);
}
const meuArray = [1, 2, 3];
loopDoWhile(meuArray);
// 1, 2, 3

This: o que é?

This: é uma palavra reservada que se refere ao objeto que está executando o código. 


 const pessoa = {
    nome: "Dio",
    idade: 5,
    cidade: "Rio de Janeiro",
    bio: function(){
        return `${this.nome}, ${this.idade} anos, ${this.cidade}`;
    }
}

pessoa.bio() // "Dio, 5 anos, Rio de Janeiro"

Seu valor pode mudar de acordo com o lugar no código onde é chamado.

Contexto Referência
Em um objeto (método) Próprio objeto dono do método
Sozinha Objeto global (em navegadores, windows)
Em uma função Global (em navegadores, windows)
Em uma função com 'use strict' Undefined
Em um evento Elemento que recebeu o evento

Manipulando seu valor

Call: chama uma função com um dado valor this e argumentos passados individualmente. 


const pessoa = {
    nome: "Dio",
    idade: 5,
    cidade: "Rio de Janeiro"   
}
const animal = {
    nome: "Gato",
    idade: 2,
    cidade: "São Paulo"
}

function bio(){
    return `${this.nome}, ${this.idade} anos, ${this.cidade}`;
}

bio.call(pessoa); // "Dio, 5 anos, Rio de Janeiro"
bio.call(animal); // "Gato, 2 anos, São Paulo"

Apply: o método apply() é usado para chamar uma função com um valor específico para isso (o contexto) e uma matriz (ou um objeto semelhante a uma matriz) de argumentos. É uma das maneiras de invocar uma função, junto com call() e bind() 


const pessoa = {
    nome: "Dio",
    idade: 5,
    cidade: "Rio de Janeiro"
}
const animal = {
    nome: "Gato",
    idade: 2,
    cidade: "São Paulo"
}

function bio(idade, cidade){
    return `${this.nome}, ${this.idade} anos, ${this.cidade}`;
}

bio.apply(pessoa, [12, "Joinville"]); // "Dio, 12 anos, Joinville"

Bind : Em JavaScript, o método bind() é usado para criar uma nova função que, quando chamada, tem um valor de this predefinido (o contexto) e, opcionalmente, um ou mais argumentos também predefinidos. 


const person = {
    firstName: 'John',
    lastName: 'Doe',
    getFullName: function() {
        return this.firstName + ' ' + this.lastName;
    }
    };
    
    const logFullName = function() {
    console.log(this.getFullName());
    };
    
    // Sem o bind, o valor de 'this' na função logFullName seria 'undefined'.
    // Ao usar bind, vinculamos o contexto do 'this' à função person.getFullName.
    const boundLogFullName = logFullName.bind(person);
    
    boundLogFullName(); // Output: "John Doe"

Arrow Functions

Arrow Functions: são funções anônimas e sempre são expressões. Não podem ser usadas como construtoras.

  1. Caso exista apenas uma linha, pode dispensar as chaves e o return.
  2. Caso exista apenas um parâmetro, pode dispensar os parênteses.

Arrow function NÃO faz hoisting. hoisting é o comportamento padrão do JavaScript de mover declarações para o topo. 


const soma = (a, b) => a + b;
soma(2, 2); // 4

const sayHello = name => `Hello ${name}`;

Restições

  1. "this"sempre será o objeto global. Métodos para modificar seu valor não irão funcionar
  2. Não existe o objeto "arguments"
  3. Não pode ser usada como construtora
